    /// <summary>
    /// Initializes the matchmaking menu and displays local player information.
    /// </summary>
    public override async void Start()
    {
        base.Start();
        if (backButton)
        {
            backButton.onClick.AddListener(() =>
            {
                StopSearchSfx();
                ss_man_PlayerInfoManager.Instance.GetMatchmakingManager.CancelMatch(OnCancelSearch =>
                {
                    OpenGameScreen();
                });
            });
        }
        p1NameText.gameObject.SetActive(false);
        p2NameText.gameObject.SetActive(false);
        var displayName = ss_man_PlayerInfoManager.Instance.GetProfileManager.ProfileModule.DisplayName;
        var avatarURL = ss_man_PlayerInfoManager.Instance.GetProfileManager.ProfileModule.Avatar.AvatarURL;
        var avatarID = ss_man_PlayerInfoManager.Instance.GetProfileManager.ProfileModule.Avatar.AvatarID;
        await ShowPlayerInfo(p1NameText, p1ProfilePic, displayName, avatarURL, avatarID);
    }
    
    /// <summary>
    /// Displays matchmaking UI for a rematch scenario with both players' information.
    /// </summary>
    public async void ShowForRematch()
    {

        var displayName = ss_man_PlayerInfoManager.Instance.GetProfileManager.ProfileModule.DisplayName;
        var avatarURL = ss_man_PlayerInfoManager.Instance.GetProfileManager.ProfileModule.Avatar.AvatarURL;
        var avatarID = ss_man_PlayerInfoManager.Instance.GetProfileManager.ProfileModule.Avatar.AvatarID;
        await ShowPlayerInfo(p1NameText, p1ProfilePic, displayName, avatarURL, avatarID);

        string LocalPlayerName = PhotonNetwork.LocalPlayer.NickName;

        string player1Name = PhotonNetwork.CurrentRoom.GetPlayer(1).NickName;
        string player2Name = PhotonNetwork.CurrentRoom.GetPlayer(2).NickName;

        bool isPlayer1Local = player1Name.Equals(LocalPlayerName);
        bool isPlayer2Local = player2Name.Equals(LocalPlayerName);
        ss_man_PlayerInfoManager.Instance.GetProfileManager.GetProfileDetail(isPlayer2Local ? player1Name : player2Name, async OnGetProfileDetails =>
        {
            if (OnGetProfileDetails.IsSuccess)
            {
                var displayName = OnGetProfileDetails.DisplayName;
                var avatarURL = OnGetProfileDetails.Avatar.AvatarURL;
                var avatarID = OnGetProfileDetails.Avatar.AvatarID;
                await ShowPlayerInfo(p2NameText, opponentProfileImage, displayName, avatarURL, avatarID);
            }
        });

        p1NameText.gameObject.SetActive(true);
        p2NameText.gameObject.SetActive(true);

        await Task.Delay(TimeSpan.FromSeconds(5f));
        StopTextureAnimation(() => { OpenGameScreen(); });
    }
    
    /// <summary>
    /// Displays player information including name and avatar.
    /// </summary>
    /// <param name="playerNameText">Text component to display the player's name.</param>
    /// <param name="playerPic">RawImage component to display the player's avatar.</param>
    /// <param name="displayName">The player's display name.</param>
    /// <param name="avatartURL">Optional URL for the player's avatar image.</param>
    /// <param name="avatartID">Optional ID for the player's avatar sprite.</param>
    async Task ShowPlayerInfo(TextMeshProUGUI playerNameText, RawImage playerPic, string displayName, string avatartURL = null, string avatartID = null)
    {
        playerNameText.SetText(displayName);
        if (avatartURL == null)
        {
            var avatar = GetAvatarByID(avatartID);
            playerPic.texture = avatar;
        }
        else
        {

            var avatar = await DownloadAvatarByURL(avatartURL);
            if (avatar != null)
            {
                playerPic.texture = avatar;
            }
            else
            {
                playerPic.texture = Config.DefaultAvatar.texture;
            }
        }
    }
    
    /// <summary>
    /// Asynchronously fetches a texture from a URL.
    /// </summary>
    /// <param name="url">The URL to fetch the texture from.</param>
    /// <param name="OnFetched">Action to invoke when the texture is fetched.</param>
    async void GetTextureFromURL(string url, Action OnFetched)
    {
        profilepicTexture = await DownloadAvatarByURL(url);
    }
    
    /// <summary>
    /// Subscribes to matchmaking status update events when enabled.
    /// </summary>
    private void OnEnable()
    {
        ss_man_CBSMatchmakingManager.OnMatchStatusUpdated += OnMatchStatusUpdated;
    }
    
    /// <summary>
    /// Unsubscribes from matchmaking status update events when disabled.
    /// </summary>
    private void OnDisable()
    {
        StopSearchSfx();
        CancelInvoke(nameof(LoadGameScene));
        isLoadingGameScene = false;
        ss_man_CBSMatchmakingManager.OnMatchStatusUpdated -= OnMatchStatusUpdated;
    }

    /// <summary>
    /// Handles matchmaking status updates, displaying opponent information when match is found.
    /// </summary>
    /// <param name="isMatchFound">Whether a match was successfully found.</param>
    /// <param name="result">The matchmaking result containing player information.</param>
    private async void OnMatchStatusUpdated(bool isMatchFound, CBSStartMatchResult result)
    {
        if (isMatchFound)
        {
            await WaitForMinimumSearchDurationAsync();
            StopSearchSfx();
            PlayMatchFoundSfx();
            await ss_man_PhotonManager.Instance.CreateOrJoinRoom($"{result.MatchID}", new ExitGames.Client.Photon.Hashtable
            {
                { "C0", $"{result.QueueName}" }
            });
            if (!await WaitForRoomReadyAsync())
            {
                OpenGameScreen();
                return;
            }
            var myProfileID = ss_man_PlayerInfoManager.Instance.GetProfileManager.ProfileModule.ProfileID;
            CBSMatchmakingPlayer opponentPlayer = null;
            foreach (var player in result.Players)
            {
                if (player.ProfileID != myProfileID)
                {
                    opponentPlayer = player;
                    break;
                }
            }
            if (opponentPlayer == null) opponentPlayer = result.Players[1];
            ss_man_PlayerInfoManager.Instance.GetProfileManager.GetProfileDetail(opponentPlayer.ProfileID, async OnGetProfileDetails =>
            {
                if (OnGetProfileDetails.IsSuccess)
                {
                    var displayName = OnGetProfileDetails.DisplayName;
                    var avatarURL = OnGetProfileDetails.Avatar.AvatarURL;
                    var avatarID = OnGetProfileDetails.Avatar.AvatarID;
                    await ShowPlayerInfo(p2NameText, opponentProfileImage, displayName, avatarURL, avatarID);
                    p1NameText.gameObject.SetActive(true);
                    p2NameText.gameObject.SetActive(true);
                    StopTextureAnimation(() =>
                    {
                        if (!isLoadingGameScene)
                        {
                            isLoadingGameScene = true;
                            Invoke(nameof(LoadGameScene), 2.25f);
                        }
                    });
                }
                else
                {
                    ss_ui_PopUp.Instance.ClosePopUp();
                    ss_ui_PopUpSettings failedPopUp = new ss_ui_PopUpSettings.Builder(PopUps.FailedPopUp, PopUpContentType.TextContent)
                        .SetTitle(LeanLocalization.GetTranslationText(LeanBakedTerms.PT_CantFindPlayer))
                        .SetButton1("OK", false, () => { ss_ui_PopUp.Instance.ClosePopUp(); OpenGameScreen(); })
                        .SetTextContent(LeanLocalization.GetTranslationText(LeanBakedTerms.PD_CantFindPlayer))
                        .SetAutoCloseTime(5f, OpenGameScreen)
                        .Build();

                    ss_ui_PopUp.Instance.OpenPopUp(failedPopUp);
                }
            });
        }
        else
        {
            StopSearchSfx();
            ss_ui_PopUp.Instance.ClosePopUp();
            ss_ui_PopUpSettings failedPopUp = new ss_ui_PopUpSettings.Builder(PopUps.FailedPopUp, PopUpContentType.TextContent)
                .SetTitle(LeanLocalization.GetTranslationText(LeanBakedTerms.PT_CantFindPlayer))
                .SetButton1("OK", false, () => { ss_ui_PopUp.Instance.ClosePopUp(); OpenGameScreen(); })
                .SetTextContent(LeanLocalization.GetTranslationText(LeanBakedTerms.PD_CantFindPlayer))
                .SetAutoCloseTime(5f, OpenGameScreen)
                .Build();

            ss_ui_PopUp.Instance.OpenPopUp(failedPopUp);
        }
    }

    /// <summary>
    /// Shows the matchmaking menu and starts the texture animation.
    /// </summary>
    /// <param name="fromDirection">The direction from which the menu appears.</param>
    /// <param name="snap">Whether to snap the menu into position immediately.</param>
    protected override void Show(enFromDirections fromDirection, bool snap)
    {
        base.Show(fromDirection, snap);

        OnAnimationStop = null;
        searchStartTimeRealtime = Time.realtimeSinceStartup;
        PlaySearchSfx();
        StartTextureAnimation();
    }
    
    /// <summary>
    /// Starts the matchmaking animation sequence with player UI sliding and avatar scrolling.
    /// </summary>
    private void StartTextureAnimation()
    {
        p1LevelBar.DOScale(0.1f, 0).OnComplete(() => { p1LevelBar.gameObject.SetActive(false); });
        p2LevelBar.DOScale(0.1f, 0).OnComplete(() => { p2LevelBar.gameObject.SetActive(false); });
        p1Coins.DOScale(0.1f, 0).OnComplete(() => { p1Coins.gameObject.SetActive(false); });
        p2Coins.DOScale(0.1f, 0).OnComplete(() => { p2Coins.gameObject.SetActive(false); });

        p1GameObject.DOAnchorPosX(-430f, 0);
        p1GameObject.DOAnchorPosX(430f, 0.5f)
            .SetEase(Ease.Linear)
            .OnComplete(() =>
            {
                p1GameObject.DOAnchorPosX(200, 0.1f)
                    .SetEase(Ease.Linear)
                    .OnComplete(() =>
                    {
                        p1GameObject.DOAnchorPosX(300, 0.5f)
                            .SetEase(Ease.OutBounce);
                    });
            });

        p2GameObject.DOAnchorPosX(430f, 0);
        p2GameObject.DOAnchorPosX(-430f, 0.5f)
            .SetEase(Ease.Linear)
            .OnComplete(() =>
            {
                p2GameObject.DOAnchorPosX(-200, 0.1f)
                    .SetEase(Ease.Linear)
                    .OnComplete(() =>
                    {
                        p2GameObject.DOAnchorPosX(-300, 0.5f)
                            .SetEase(Ease.OutBounce);
                    });
            });

        opponentProfileImage.uvRect = new Rect(0, 0, 1, 0.125f);
        opponentProfileImage.texture = testAvatarsListTexture;
        playerFoundImage.gameObject.SetActive(false);
        playerFoundImage.transform.DOScale(0.75f, 0);

        scrollTween = DOTween.To(() => opponentProfileImage.uvRect.y,
                   y => opponentProfileImage.uvRect = new Rect(opponentProfileImage.uvRect.x, y, opponentProfileImage.uvRect.width, opponentProfileImage.uvRect.height),
                   1,
                   scrollSpeed)
               .SetEase(Ease.Linear)
               .SetLoops(-1, LoopType.Restart)
               .OnKill(() =>
               {
                   CompleteAnimation();
               });
    }

    /// <summary>
    /// Completes the matchmaking animation by showing player info and "Player Found" indicator.
    /// </summary>
    private void CompleteAnimation()
    {
        p1LevelBar.gameObject.SetActive(true);
        p2LevelBar.gameObject.SetActive(true);
        p1Coins.gameObject.SetActive(true);
        p2Coins.gameObject.SetActive(true);

        p1LevelBar.DOScale(1f, 0.75f).SetEase(Ease.OutBounce).OnComplete(() =>
        {
            p1Coins.DOScale(1f, 0.5f).SetEase(Ease.OutBounce);
            p1CoinsAnimator.Play($"CoinsAnimation");
        });

        p2LevelBar.DOScale(1f, 0.75f).SetEase(Ease.OutBounce).OnComplete(() =>
        {
            p2Coins.DOScale(1f, 0.5f).SetEase(Ease.OutBounce);
            p2CoinsAnimator.Play($"CoinsAnimation");
        });


        playerFoundImage.gameObject.SetActive(true);

        playerFoundImage.color = new Color(playerFoundImage.color.r, playerFoundImage.color.g, playerFoundImage.color.b, 0f);
        opponentProfileImage.uvRect = new Rect(0, 0, 1, 1);
        playerFoundImage.transform.DOScale(1.3f, 0.5f);
        playerFoundImage.DOFade(1f, 0.25f).OnComplete(() =>
        {
            //opponentProfileImage.texture = testOpponentAvatar;

            playerFoundImage.DOFade(0f, 0.5f).OnComplete(() =>
            {
                playerFoundImage.gameObject.SetActive(false);
                playerFoundImage.transform.localScale = Vector3.one;
                OnAnimationStop?.Invoke();
            });
        });
    }

    [Header("Animation Events")]
    [Tooltip("Action invoked when the texture animation stops.")]
    public static Action OnAnimationStop = null;
    
    /// <summary>
    /// Stops the texture scrolling animation and invokes the stop callback.
    /// </summary>
    /// <param name="_OnAnimationStop">Optional callback to invoke when animation stops.</param>
    public void StopTextureAnimation(Action _OnAnimationStop = null)
    {
        if (scrollTween != null && scrollTween.IsActive())
        {
            if (_OnAnimationStop != null) OnAnimationStop = _OnAnimationStop;
            scrollTween.Kill();
        }
    }
    
    /// <summary>
    /// Returns to the main game screen by hiding this menu and showing the main menu.
    /// </summary>
    void OpenGameScreen()
    {
        StopSearchSfx();
        this.HideToLeft();
        mainMenu.JustShow();
    }
    
    /// <summary>
    /// Loads the match scene when a match is found and ready.
    /// </summary>
    void LoadGameScene()
    {
        if (!PhotonNetwork.InRoom || PhotonNetwork.CurrentRoom == null || PhotonNetwork.CurrentRoom.PlayerCount < 2)
        {
            isLoadingGameScene = false;
            return;
        }

        //matchScene.Open().Close(AdvancedSceneManager.SceneManager.openScenes).WithLoadingScreen(true);
        SceneManager.LoadScene(matchScene);
    }

    /// <summary>
    /// Downloads an avatar texture from a URL.
    /// </summary>
    /// <param name="avatarURL">The URL of the avatar image to download.</param>
    /// <returns>The downloaded texture, or default avatar if download fails.</returns>
    async Task<Texture2D> DownloadAvatarByURL(string avatarURL)
    {
        Texture2D texture = await ss_man_PlayfabManager.Instance.DownloadImageTexture(avatarURL);
        if (texture == null)
        {
            return Config.DefaultAvatar.texture;
        }
        return texture;
    }
    
    /// <summary>
    /// Gets an avatar texture by its ID from the avatar icons data.
    /// </summary>
    /// <param name="avatarID">The ID of the avatar to retrieve.</param>
    /// <returns>The avatar texture, or default avatar if not found.</returns>
    Texture2D GetAvatarByID(string avatarID)
    {
        var spriteAvatar = AvatarsData.GetSprite(avatarID);
        if (spriteAvatar == null)
        {
            return Config.DefaultAvatar.texture;
        }
        else
        {
            return spriteAvatar.texture;
        }
    }

    private async Task WaitForMinimumSearchDurationAsync()
    {
        float elapsedSeconds = Time.realtimeSinceStartup - searchStartTimeRealtime;
        float remainingSeconds = minSearchDurationSeconds - elapsedSeconds;
        if (remainingSeconds <= 0f)
        {
            return;
        }

        int delayMs = Mathf.CeilToInt(remainingSeconds * 1000f);
        await Task.Delay(delayMs);
    }

    private void PlaySearchSfx()
    {
        var audioManager = AudioManager.TryGetInstance();
        if (audioManager == null)
        {
            return;
        }

        audioManager.PlaySFX(MatchSearchClipName, loop: true, stopIfLooping: true);
    }

    private void StopSearchSfx()
    {
        var audioManager = AudioManager.TryGetInstance();
        if (audioManager == null)
        {
            return;
        }

        audioManager.StopSFX(MatchSearchClipName);
    }

    private void PlayMatchFoundSfx()
    {
        var audioManager = AudioManager.TryGetInstance();
        if (audioManager == null)
        {
            return;
        }

        audioManager.PlaySFX(MatchFoundClipName);
    }

    private async Task<bool> WaitForRoomReadyAsync()
    {
        int elapsedMs = 0;
        while (elapsedMs < RoomReadyTimeoutMs)
        {
            if (PhotonNetwork.InRoom && PhotonNetwork.CurrentRoom != null && PhotonNetwork.CurrentRoom.PlayerCount >= 2)
            {
                return true;
            }

            await Task.Delay(RoomReadyPollDelayMs);
            elapsedMs += RoomReadyPollDelayMs;
        }

        Debug.LogWarning("Timed out waiting for Photon room to become ready.");
        return false;
    }
